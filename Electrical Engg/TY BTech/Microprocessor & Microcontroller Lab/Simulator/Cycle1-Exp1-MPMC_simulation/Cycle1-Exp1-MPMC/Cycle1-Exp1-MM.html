Block Exchange and Block transfer using 8051

ASSEMBLY CODE FOR BLOCK EXCHANGE

ORG 000h
mov r0, #30h
mov r1,#40h
mov r7,#05h
repeat: mov a,@r0
mov r4,a
mov a,@r1
mov @r0,a
mov a,r4
mov @r1,a
inc r0
inc r1
djnz r7 repeat
end
